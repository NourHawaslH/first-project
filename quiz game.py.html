    print('correct!')
    score += 1
else:
    print("incorrect!")

answer = input("what does ram stand for? ")
if answer.lower() == 'random access memory':
    print('correct!')
    score += 1
else:
    print("incorrect!")

if answer.lower() == "power supply":
    print('correct!')
    score += 1
else:
    print("incorrect!")
